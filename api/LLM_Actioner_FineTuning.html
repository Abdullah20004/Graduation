import pandas as pd
import os
import torch
import gc
from trl import SFTConfig, SFTTrainer

# 0. CLEAN VRAM BEFORE STARTING
print("Cleaning VRAM...")
gc.collect()
torch.cuda.empty_cache()
if torch.cuda.is_available():
    torch.cuda.reset_peak_memory_stats()
    print(f"Initial VRAM usage: {torch.cuda.memory_allocated() / 1024**2:.2f} MB")

from transformers import AutoModelForCausalLM, AutoTokenizer, BitsAndBytesConfig, TrainingArguments, TrainerCallback
from peft import LoraConfig, get_peft_model, prepare_model_for_kbit_training
from datasets import load_dataset

MODEL_NAME = "mickelliu/Self-RedTeam-Qwen2.5-7B-Instruct"
OUTPUT_DIR = "checkpoints/red_actioner_7b"
LOG_DIR = "logs/red_actioner"

os.makedirs(LOG_DIR, exist_ok=True)
output_prefix = "data/red_actioner_sft"

# 1. Load Data
dataset = load_dataset("json", data_files={"train": f"{output_prefix}_train.json", "test": f"{output_prefix}_eval.json"})

# 2. Quantization Config - Switched to BFloat16 for RTX 4060
bnb_config = BitsAndBytesConfig(
    load_in_4bit=True,
    bnb_4bit_quant_type="nf4",
    bnb_4bit_compute_dtype=torch.bfloat16, # <-- Changed to bfloat16
    bnb_4bit_use_double_quant=True,
)

# 3. Model & Tokenizer
tokenizer = AutoTokenizer.from_pretrained(MODEL_NAME, token=token)
tokenizer.pad_token = tokenizer.eos_token

print("Loading model to GPU...")
model = AutoModelForCausalLM.from_pretrained(
    MODEL_NAME, 
    quantization_config=bnb_config, 
    device_map={"": 0}, 
    token=token
)
model = prepare_model_for_kbit_training(model)

# 4. PEFT Config
peft_config = LoraConfig(
    lora_alpha=32, r=16, lora_dropout=0.05, task_type="CAUSAL_LM",
    target_modules=["q_proj", "k_proj", "v_proj", "o_proj", "gate_proj", "up_proj", "down_proj"]
)
model = get_peft_model(model, peft_config)

class ResultsCallback(TrainerCallback):
    def on_log(self, args, state, control, logs=None, **kwargs):
        if logs and "loss" in logs:
            pd.DataFrame([logs]).assign(step=state.global_step).to_csv(f"{LOG_DIR}/results.csv", mode='a', header=not os.path.exists(f"{LOG_DIR}/results.csv"), index=False)

# 5. Training Args - Using BF16 for RTX 4060 compatibility
args = SFTConfig(
    output_dir=OUTPUT_DIR, 
    num_train_epochs=3, 
    per_device_train_batch_size=1, 
    gradient_accumulation_steps=4, 
    learning_rate=2e-4, 
    bf16=True,                       # <-- Changed to True
    fp16=False,                      # <-- Changed to False
    logging_steps=1,                 
    eval_strategy="steps", 
    eval_steps=5,                   
    save_steps=10,
    logging_strategy="steps",
    report_to="none",                
    lr_scheduler_type="cosine",
    max_grad_norm=0.3,
    disable_tqdm=False,
    dataset_text_field="messages",
    max_length=512
)

trainer = SFTTrainer(
    model=model, 
    train_dataset=dataset["train"], 
    eval_dataset=dataset["test"],
    args=args,
    processing_class=tokenizer,
    callbacks=[ResultsCallback()]
)

print("Starting fine-tuning. Monitor the progress bar below for Loss and Learning Rate.")
trainer.train()
trainer.model.save_pretrained(f"{OUTPUT_DIR}/final_lora")

from peft import PeftModel
import torch

def test_base_model(url, context_signal):
    """Runs the prompt against the BASE model by disabling the LoRA adapters."""
    
    # Prepare the prompt exactly as before
    prompt = tokenizer.apply_chat_template([
        {"role": "system", "content": "You are HackOps Red Actioner. Think in <think> tags."},
        {"role": "user", "content": f"Target: {url}, Signal: {context_signal}"}
    ], tokenize=False, add_generation_prompt=True)
    
    inputs = tokenizer(prompt, return_tensors="pt").to("cuda")
    
    # We use 'disable_adapter' to temporarily ignore our training
    print(f"--- RUNNING TEST ON BASE MODEL (No Fine-Tuning) ---")
    with torch.no_grad():
        with model.disable_adapter():
            outputs = model.generate(
                **inputs, 
                max_new_tokens=1024, 
                temperature=0.3,
                top_p=0.9,
                repetition_penalty=1.1
            )
    
    print(tokenizer.decode(outputs[0], skip_special_tokens=False))

# Execute the comparison
